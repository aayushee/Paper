\begin{algorithm}
  \KwIn{Ocr.txt,Corrected.txt,Original.txt}
  \KwOut{Algorithm Accuracy }
  Ocrline:=a line of text from Ocr.txt\;
 Correctedline:=a line of text from Corrected.txt\;
 Originalline:=a line of text from Original.txt\;
  \For{ocrstring:Ocrline}{
    \If{(ocrstring==correctedstring)}
	{
      	   \If(\tcp*[h]{If length of ocrline or original line is less than 4})
	{Ocrline.length $<$4 $||$ Originalline.length$<$ 4}  
	   {
     	     \For{(originalstring:originalline)}
	         {
		\If (\tcp*[h]{If ocr string matches any string in original line}){(ocrstring==originalstring)}
		{
		truenegative+=1\;
		flag=false; break\;
		 }
		}
	     \If{(flag==true)}
		{falsenegative+=1\;}
	   }
	\Else 
		{  (\tcp*[h]{if ocr and original lines are of length greater than 4})
      		
		\If (\tcp*[h]{If ocrstring is at index 0}){(i==0)}
		{
			\For{(int j=0 to 2)}
			{
		 	\If(\tcp*[h]{If ocrstring matches any string in original line from index window 1 to 3}){(ocrline[i]==originalline[j])}
		  	{
				truenegative+=1\;
				flag=false; break\;
		 	}}
			\If{(flag==true)}
			{falsenegative+=1\;
			}
		}
		\If(\tcp*[h]{If ocr string is at index 1}){(i==1)}
		{
			\For{(int j=0 to 3)}
			{
				\If(\tcp*[h]{If ocrstring matches any string in original line from index window 1 to 4}){(ocrline[i]==originalline[j])}
				{
					truenegative+=1\;
					flag=false;break\;
				}
			}
			\If{(flag==true)}
			{falsenegative+=1\;}
		}
\If(\tcp*[h]{If there is difference in lengths of original and ocr line})
{(i==originalline.length-2 $||$ originalline.length-1 $||$ originalline.length $||$ originalline.length+1)}
		{
			\For{(j=i-2 to originalline.length)}
			{
				\If(\tcp*[h]{If ocrstring matches any strings uptil length of original line})
				{(ocrline[i]==originalline[j])}
				{
					truenegative+=1\;
					flag=false;break\;
				}
			}
			\If{(flag==true)}
			{falsenegative+=1\;}
		}
		\If(\tcp*[h]{If ocrstring is longer than 2 index window of original line})
		{(i$>$= originalline.length+2)}
		{
			\For{(j=originalline.length-3 to originalline.length)}
			{
				\If(\tcp*[h]{If ocrstring matches any strings at last 3 indices in original line})
				{(ocrline[i]==originalline[j])}
				{
					truenegative+=1\;
					flag=false;break\;
				}
			}
			\If{(flag==true)}
			{falsenegative+=1\;}
		}
	
	}
	}
      }
    
  \caption{Edit distance algorithm evaluation}
\end{algorithm}
